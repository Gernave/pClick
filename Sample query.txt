Select ((Firs.tbs*100)/second.tbs) From
(Select Distinct Count(question_id) as tbs from votes A 
inner join questions c ON
c.id=A.question_id
where A.ans="F" and c.id="5" )as firs,
(Select Distinct Count(question_id) as tbs from votes A 
inner join questions c ON
c.id=A.question_id
 where c.id ="5"
) as second;
